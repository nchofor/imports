// Fade-in on scroll const faders = document.querySelectorAll('.card, .blog-card, .hero-content'); const appearOptions = { threshold: 0.1, rootMargin: "0px 0px -50px 0px" }; const appearOnScroll = new IntersectionObserver(function(entries, observer){ entries.forEach(entry => { if (!entry.isIntersecting) return; entry.target.classList.add('appear'); observer.unobserve(entry.target); }); }, appearOptions); faders.forEach(fader => { appearOnScroll.observe(fader); });
